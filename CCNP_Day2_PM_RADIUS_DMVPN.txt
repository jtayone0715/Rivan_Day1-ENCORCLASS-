2. policy

COMMAND
!@UTM-US - HUB
conf t
 crypto ikev2 proposal IKEV2-PROP
  encryption aes-cbc-256
  integrity sha256
  group 14
 !
 crypto ikev2 proposal IKEV2-PROP2
  encryption aes-gcm-256
  prf sha256
  group 14
!
 crypto ikev2 policy IKEV2-POL
  proposal IKEV2-PROP
  proposal IKEV2-PROP2
 !
 crypto ikev2 keyring KEY
  peer SPOKES
   address 0.0.0.0 0.0.0.0
   pre-shared-key keykeymo
!
 crypto ikev2 profile IKEV2-PROF
  match identity remote any
  authentication remote pre-share
  authentication local pre-share
  keyring local KEY
  end

IKEV2 PHASE 2
!@UTM-US - HUB
conf t
 crypto ipsec transform-set TS1 esp-aes 256 esp-sha256-hmac
  mode tunnel
 crypto ipsec transform-set TS2 esp-gcm 256
  mode tunnel
 !
 crypto ipsec profile DMVPN-IPSEC
  set transform-set TS1 TS2
  set ikev2-profile IKEV2-PROF
 !
 interface Tunnel1
  ip add 172.16.0.100 255.255.255.0
  no shut
  tunnel source g1
  tunnel mode gre multipoint
  tunnel protection ipsec profile DMVPN-IPSEC
  !
  ip nhrp network-id 100
  ip nhrp map multicast dynamic
  ip nhrp authentication dmvpn
  ip nhrp redirect
  end
